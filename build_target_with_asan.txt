Bash commands to execute at the root project directory :

for GCC

cmake -S . -B build-asan \
  -DENABLE_ASAN=ON \
  -DCMAKE_BUILD_TYPE=Debug \
  -DCMAKE_PREFIX_PATH=$HOME/Qt/6.10.1/gcc_64

cmake --build build-asan

for CLANG

cmake -S . -B build-asan-clang \
  -DENABLE_ASAN=ON \
  -DCMAKE_BUILD_TYPE=Debug \
  -DCMAKE_CXX_COMPILER=clang++ \
  -DCMAKE_C_COMPILER=clang \
  -DCMAKE_PREFIX_PATH=$HOME/Qt/6.10.1/gcc_64

cmake --build build-asan-clang


build-asan/src/app/Ofeli

# commande de base pour GCC, sans filtre mais avec réorganisation par symbole des messages (option symbolize=1)

ASAN_OPTIONS=halt_on_error=1:detect_leaks=1:symbolize=1:strict_string_checks=1:check_initialization_order=1:detect_stack_use_after_return=1 build-asan/src/app/Ofeli

# utiliser clang pour filtrer les messages de Qt

ASAN_OPTIONS=halt_on_error=1:symbolize=1:strict_string_checks=1:check_initialization_order=1:detect_stack_use_after_return=1 \
LSAN_OPTIONS=detect_leaks=1:suppressions=$(pwd)/lsan.supp \
build-asan-clang/src/app/Ofeli
